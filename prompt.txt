buat tabel users:
id integer auto increment
name varchar 255 not null

created_at timestamp default current_timestamp

    "name": "eko",
    "email": "eko@localhost.com",
    "password": "rahasia"
}

}

{
    "error": "email sudah terdaftar"
}
